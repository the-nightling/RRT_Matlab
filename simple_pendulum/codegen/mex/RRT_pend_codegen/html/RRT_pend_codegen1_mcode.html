1function control = RRT_pend_codegen
 2%#codegen
 3% Computes the sequence of control actions needed for the swing up of
 4% a simple pendulum using a Rapidly Exploring Random Tree.
 5% The tree will grow from the initial state and span the phase space area,
 6% looking for the goal point/state (indicated by the red marker).
 7% If the goal state was located, a red line traces back the sequence of states used
 8% to reach the goal state.
 9
10    x0 = [-pi/2; 0];    % initial state; angle position measured from x-axis
11    xG = [pi/2; 0];     % goal state
12    goalRadiusSq = 0.02;    % distance from goal that is an acceptable solution
13    goalBias = 0.99;
14    xlimits = [-pi,pi; -10,10]; % state limits
15    U = linspace(-5,5,20);      % range of control torques that can be used
16    dt = 0.1;           % time interval between application of subsequent control torques
17    
18    u_path = 0;
19    
20    N = 20000;  % maximum number of iterations
21
22    % pre-allocating memory
23    G = repmat(x0,1,N);     % graph of states in RRT
24    P = ones(1,N);          % parent states
25    Ui = ones(1,N);         % control actions
26    xn_c = repmat([1;1],1,length(U));
27
28    % keep growing RRT util goal found or run out of iterations
29    for n = 2:N
30    
31        % get random state
32        bUseGoal = rand(1)>goalBias;
33        xn = ~bUseGoal.*rand(2,1).*(xlimits(:,2)-xlimits(:,1)) + xlimits(:,1);
34        
35        % find distances between that state point and every vertex in RRT
36        dsq = euclidianDistSquare(xn,G(:,1:n));
37        
38        % select RRT vertex closest to the state point
39        [~,i] = min(dsq);
40        
41        % from the closest RRT vertex, compute all the states that can be reached,
42        % given the pendulum dynamics and available torques
43        for ui = 1:length(U)
44            xn_c(:,ui) = G(:,i) + dt*dynamics(G(:,i),U(ui));
45        end
46        
47        % select the closest reachable state point
48        dsq = euclidianDistSquare(xn,xn_c);
49        [~,ui] = min(dsq);
50        xn = xn_c(:,ui);
51        
52        % if angular position is greater than pi rads, wrap around
53        xn(1) = mod(xn(1)+pi,2*pi)-pi;
54                
55        % link reachable state point to the nearest vertex in the tree
56        G(:,n) = xn;
57        P(1,n) = i;
58        Ui(1,n) = ui;
59
60        % if the goal was reached,
61        % retrace steps from goal state to initial state
62        % path displayed using red line
63        if(sum((xG-xn).^2,1) < goalRadiusSq)
64            xbi = 0;
65            
66            % retrace control actions and solution trajectory
67            u_path = [];
68            while(xbi(1) ~= 1)              
69                u_path = [U(Ui(xbi(1))),u_path];
70                xbi = [P(xbi(1)),xbi];
71            end
72
73            break;
74        end
75    end
76    
77    control = [[0:dt:dt*length(u_path)-dt]',u_path'];
78    assignin('base', 'control', control);
79end
